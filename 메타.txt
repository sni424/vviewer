      바닥1, 바닥2

    DP_ON: [
      디피, 벽, 바닥, 화장실, 거실
    ],
    DP_OFF: [
      벽, 바닥, 화장실, 거실
    ]
  },
  클리핑 : {
    y: 1.5,
    모델 : glb
  }
  투어: [
    {
      인덱스: 1,
      매트릭스: [1,0,0, ...], // 16자리 카메라 RT매트릭스
      이름: 거실
    }, ...
  ]
  포스트프로세싱 : {
    블룸:{},
    밝기/대비:{},
  },
  옵션:[
    {
      옵션위치 : [x,y,z],
      관찰자매트릭스: [1,0,0, ...], // 16자리 RT매트릭스
      위치: [거실, 방1]

    }, ...
  ],
  방:[
    {

        [0,0,1],
